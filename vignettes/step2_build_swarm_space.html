2.1 Load data

Load the metrics of collective motion calculated previously. Include the events you want to compare in a swarm space.

library(swarmyverse)
data('multi_species_metrics')

all_data <- multi_species_metrics
## or bind with new data
# all_data <- rbind(multi_species_metrics, swarm_space_metrics)

2.2 Build swarm space

Create new swarm space with PCA analysis:

new_pca <- create_swarm_space(metrics_data = all_data, 
                              space_type = "pca",
                              save_pca_object = FALSE)

ggplot2::ggplot(new_pca, ggplot2::aes(x = PC1, y = PC2, color = species)) +
  ggplot2::geom_point() +
  ggplot2::theme_bw()

Or create a new swarm space with tSNE:

new_tsne <- create_swarm_space(metrics_data = all_data,
                              space_type = "tsne",
                              tsne_rand_seed = 2022,
                              tsne_perplexity = 10
                              )
