1. Load and transform data - trackdf

library(swarmyverse)

raw <- read.csv(system.file("extdata/video/01.csv", package = "trackdf"))
raw <- raw[!raw$ignore, ]

data_df <- set_data_format(raw_x = raw$x,
                          raw_y = raw$y,
                          raw_t = raw$frame,
                          raw_id = raw$track,
                          origin = "2019-03-24 12:55:23",
                          period = "1 sec",
                          tz = "America/New_York",
                          )

data_df <- add_velocities_parallel(data_df, 
                                  sample_step = 10,
